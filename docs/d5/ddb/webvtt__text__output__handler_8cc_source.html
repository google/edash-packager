34 Status WebVttTextOutputHandler::Process(
35  std::unique_ptr<StreamData> stream_data) {
36  switch (stream_data->stream_data_type) {
37  case StreamDataType::kStreamInfo:
38  return OnStreamInfo(*stream_data->stream_info);
39  case StreamDataType::kSegmentInfo:
40  return OnSegmentInfo(*stream_data->segment_info);
41  case StreamDataType::kCueEvent:
42  return OnCueEvent(*stream_data->cue_event);
43  case StreamDataType::kTextSample:
44  OnTextSample(*stream_data->text_sample);
45  return Status::OK;
46  default:
47  return Status(error::INTERNAL_ERROR,
48  "Invalid stream data type for this handler");
49  }
50 }

72 Status WebVttTextOutputHandler::OnSegmentInfo(const SegmentInfo& info) {
73  total_duration_ms_ += info.duration;
74 
75  const std::string& segment_template = muxer_options_.segment_template;
76  const uint32_t index = segment_index_++;
77  const uint64_t start = info.start_timestamp;
78  const uint64_t duration = info.duration;
79  const uint32_t bandwidth = muxer_options_.bandwidth;
80 
81  const std::string filename =
82  GetSegmentName(segment_template, start, index, bandwidth);
83 
84  // Write everything to the file before telling the manifest so that the
85  // file will exist on disk.
86  std::unique_ptr<File, FileCloser> file(File::Open(filename.c_str(), "w"));
87 
88  if (!file) {
89  return Status(error::FILE_FAILURE, "Failed to open " + filename);
90  }
91 
92  buffer_.WriteTo(file.get());
93  buffer_.Reset();
94 
95  if (!file.release()->Close()) {
96  return Status(error::FILE_FAILURE, "Failed to close " + filename);
97  }
98 
99  // Update the manifest with our new file.
100  const uint64_t size = File::GetFileSize(filename.c_str());
101  muxer_listener_->OnNewSegment(filename, start, duration, size);
102 
103  return Status::OK;
104 }
